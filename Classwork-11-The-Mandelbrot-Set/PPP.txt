    Initialize an empty dictionary named 'config'
    
    Open 'config.txt' in read mode as 'file'
    For each 'line' in 'file':
        Remove leading/trailing whitespaces from 'line'
        Split 'line' by the character "=" into 'parameter' and 'value'
        
        If 'value' contains '.':
            Convert 'value' to a floating-point number
        Else:
            Convert 'value' to an integer
        EndIf
        
        Store the converted value in 'config' under the key 'parameter'
    EndFor
    Close 'file'

    #Assign configuration variables
    Set 'width' to config['ancho']
    Set 'height' to config['alto']
    Set 'max_iter' to config['max_iter']

    #Initialize output file
    Open 'mandelbrot.csv' in write mode as 'output'
    Write the header line 'row,column,iterations\n' to 'output'

    #Generate Mandelbrot set data
    For 'row' from 0 to 'height' - 1:
        For 'column' from 0 to 'width' - 1:
            
            Set 'real' to config['real_min'] + (column / width) * (config['real_max'] - config['real_min'])
            Set 'imag' to config['imag_min'] + (row / height) * (config['imag_max'] - config['imag_min'])
            
            Create a complex number 'c' using ('real', 'imag')
            Initialize complex number 'z' to 0 + 0i
            Initialize 'iterations' to 0
        
            While (absolute value of 'z' <= 2) AND ('iterations' < 'max_iter'):
                Set 'z' to (z * z) + c
                Increment 'iterations' by 1
            EndWhile
            
            // Write results to CSV
            Write 'row', 'column', and 'iterations' separated by commas to 'output'
            
        EndFor
    EndFor

    Close 'output'
